<h1>Four in a Row Game</h1>
<div id="game-board"></div>
<button id="reset">Reset Game</button>
<div id="message">Player 1's turn (Red)</div>
<script>
    const rows = 6;
    const cols = 7;
    let currentPlayer = 'player1'; // Start with Player 1
    let gameBoard = Array.from({ length: rows }, () => Array(cols).fill(null)); // Create an empty game board
    const gameBoardElement = document.getElementById('game-board');
    const resetButton = document.getElementById('reset');
    const messageElement = document.getElementById('message');
    function drawBoard() {
        gameBoardElement.innerHTML = ''; // Clear the game board
        for (let row = 0; row < rows; row++) {
            for (let col = 0; col < cols; col++) {
                const cell = document.createElement('div');
                cell.classList.add('cell');
                if (gameBoard[row][col]) {
                    cell.classList.add(gameBoard[row][col]); // Apply color based on the player
                }
                cell.addEventListener('click', () => dropDisc(col));
                gameBoardElement.appendChild(cell);
            }
        }
    }
    function dropDisc(col) {
        // Find the first available row in the selected column
        for (let row = rows - 1; row >= 0; row--) {
            if (!gameBoard[row][col]) {
                gameBoard[row][col] = currentPlayer; // Drop the disc
                drawBoard();
                if (checkWinner(row, col)) {
                    messageElement.textContent = `${currentPlayer === 'player1' ? 'Player 1' : 'Player 2'} wins!`;
                    setTimeout(resetGame, 2000); // Reset the game after 2 seconds
                } else {
                    currentPlayer = (currentPlayer === 'player1') ? 'player2' : 'player1'; // Switch player
                    messageElement.textContent = `${currentPlayer === 'player1' ? 'Player 1' : 'Player 2'}'s turn`;
                }
                break;
            }
        }
    }
    function checkWinner(row, col) {
        // Check for 4 in a row in multiple directions
        return (
            checkDirection(row, col, 1, 0) || // Horizontal
            checkDirection(row, col, 0, 1) || // Vertical
            checkDirection(row, col, 1, 1) || // Diagonal /
            checkDirection(row, col, 1, -1)   // Diagonal \
        );
    }
    function checkDirection(row, col, rowDir, colDir) {
        let count = 1;
        // Check one direction (e.g., right or up-right)
        for (let i = 1; i < 4; i++) {
            const r = row + rowDir * i;
            const c = col + colDir * i;
            if (r >= 0 && r < rows && c >= 0 && c < cols && gameBoard[r][c] === currentPlayer) {
                count++;
            } else {
                break;
            }
        }
        // Check the opposite direction (e.g., left or down-left)
        for (let i = 1; i < 4; i++) {
            const r = row - rowDir * i;
            const c = col - colDir * i;
            if (r >= 0 && r < rows && c >= 0 && c < cols && gameBoard[r][c] === currentPlayer) {
                count++;
            } else {
                break;
            }
        }
        return count >= 4; // If there are 4 discs in a row
    }
    function resetGame() {
        gameBoard = Array.from({ length: rows }, () => Array(cols).fill(null)); // Reset the board
        currentPlayer = 'player1'; // Reset the starting player
        messageElement.textContent = `Player 1's turn (Red)`; // Reset message
        drawBoard();
    }
    resetButton.addEventListener('click', resetGame);
    drawBoard(); // Initial board drawing
</script>
